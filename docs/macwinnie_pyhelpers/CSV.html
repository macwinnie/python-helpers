 13def xlsCol2Int(colName):
 14    """XLS col to int
 15
 16    According to `A` is `0`, `Z` is `26`, `AA` is `27` and so on, this
 17    method is meant to translate the alphabetic “number” to an integer.
 18
 19    Args:
 20        colName (str): XLS column representation, e.g. `A` or `AA`, ...
 21
 22    Returns:
 23        int: index representation as integer
 24    """
 25    val = 0
 26    for ch in colName:  # base-26 decoding "+1"
 27        val = val * 26 + ord(ch) - ord0 + 1
 28    return val - 1

 31def int2xlsCol(colInt):
 32    """int index to XLS index
 33
 34    According to `A` is `0`, `Z` is `26`, `AA` is `27` and so on, this
 35    function is meant to translate an integer to its alphabetic “number”
 36    representation.
 37
 38    Args:
 39        colInt (int): index to be transferred to XLS column representation
 40
 41    Returns:
 42        str: XLS column representation
 43    """
 44    chars = []
 45    while True:
 46        if len(chars) > 0:
 47            colInt = colInt - 1
 48        ch = colInt % 26
 49        chars.append(chr(ch + ord0))
 50        colInt = colInt // 26
 51        if not colInt:
 52            break
 53    return "".join(reversed(chars))

 73            "emptyValue": None,
 74            "linebreak": "\n",
 75        }

 77        self.rows = []
 78        self.rowCheck = False
 79        self.loadData(data, kvDict)

 94    def __str__(self):
 95        """convert CSV object into CSV string
 96
 97        Returns:
 98            str: CSV string
 99        """
100        self.refreshFromRows()
101
102        combined = [[k] + v for k, v in self.data.items()]
103
104        cols = pd.DataFrame(combined).T.values.tolist()
105        rows = []
106
107        for r in cols:
108            i = 0
109
110            for c in r:
111                if c == self.specs["emptyValue"]:
112                    r[i] = ""
113                i += 1
114            rows.append(
115                '"{values}"'.format(
116                    values='"{delim}"'.format(delim=self.specs["delimiter"]).join(
117                        [
118                            (
119                                str(c).replace('"', '""')
120                                if isinstance(c, str) or not math.isnan(c)
121                                else ""
122                            )
123                            for c in r
124                        ]
125                    )
126                )
127            )
128
129        return self.specs["linebreak"].join(rows) + self.specs["linebreak"]

169    def loadData(self, data, kvDict=False, emptyValue=None, skipRows=False):
170        """load data
171
172        Method to load the data of the CSV.
173        If data is a list of dicts, the dicts should have the keys in
174        common and `kvDict` shall be `True`.
175        If the data is a dict of column-keys with a list of column-values
176        where the index is the number of the row, `kvDict` shall be `False`.
177
178        Args:
179            data (mixed):       CSV data
180            kvDict (bool):      is the data key-value structured? (default: `False`)
181            emptyValue (mixed): value to override `self.specs['emptyValue']` to be
182                                used for empty columns
183            skipRows (bool):    should the data be transferred into rows? DON'T USE
184                                UNLESS YOU KNOW WHAT YOU'RE DOING! (default: `False`)
185        """
186        self.rowCheck = False
187
188        if emptyValue == None:
189            emptyValue = self.specs["emptyValue"]
190
191        if kvDict:
192            self.data = {}
193            for row in data:
194                try:
195                    existingRowCount = len(next(iter(self.data.values())))
196                except:
197                    existingRowCount = 0
198                keys = row.keys()
199                for k in keys:
200                    if k not in self.data:
201                        if existingRowCount > 0:
202                            self.data[k] = [emptyValue] * existingRowCount
203                        else:
204                            self.data[k] = []
205                for k in self.data.keys():
206                    if k in row:
207                        self.data[k].append(row[k])
208                    else:
209                        self.data[k].append(emptyValue)
210        else:
211            self.data = data
212
213        if not skipRows:
214            self.getRows(force=True)

298        """reload the CSV from rows
299
300        When one iterates over the rows of a CSV class and changes values, e.g.
301        adding new columns or manipulate values, the CSV object needs to be
302        rebuilt from the rows ... that's what this method is doing.
303        """
304        if self.rowCheck:
305            self.loadData(data=self.rows, kvDict=True, skipRows=True)
306            self.rowCheck = False

308    def dropEmptyColumns(self, ensureKeyEquality=True, emptyValue=None):
309        """drop empty columns
310
311        sometimes it's usefull to clean up a CSV file and drop empty columns,
312        e.g. if the CSV was created by an export
313        """
314        self.getRows(force=True, keepEmpty=False)
315
316        if emptyValue == None:
317            emptyValue = self.specs["emptyValue"]
318
319        if ensureKeyEquality:
320            keys = []
321            for r in self.rows:
322                keys = list(set(keys) | set(r.keys()))
323            for i in range(0, len(self.rows)):
324                for k in keys:
325                    r = self.rows[i]
326                    if k not in r:
327                        r[k] = emptyValue
328
329        self.rowCheck = True
330        self.refreshFromRows()

332    def keyValueRows(self, keepEmpty=False, emptyValue=None):
333        """data to list of rows
334
335        Method to get row representation of CSV.
336
337        Args:
338            keepEmpty (bool): if set to True, empty values are kept in row representation (default: `False`)
339            emptyValue (mixed): value that should be used for kept empty values (default: `None`)
340
341        Returns:
342            list: list of rows in CSV
343        """
344        if emptyValue == None:
345            emptyValue = self.specs["emptyValue"]
346
347        self.refreshFromRows()
348
349        kvrows = []
350        keys = list(self.data.keys())
351        if len(keys) > 0:
352            count = len(self.data[keys[0]])
353            i = 0
354            while i < count:
355                row = {}
356                for key in keys:
357                    value = self.data[key][i]
358                    if (
359                        value == ""
360                        or (type(value) == float and math.isnan(value))
361                        or value == emptyValue
362                    ):
363                        if keepEmpty:
364                            row[key] = emptyValue
365                    else:
366                        row[key] = value
367                kvrows.append(row)
368                i += 1
369        return kvrows

371    def writeFile(self, filepath, delimiter=None, linebreak=None, backupExisting=False):
372        """write CSV file
373
374        Method to write out data of current object to a CSV file.
375
376        Args:
377            filepath (str): path to destination file
378            delimiter (str): delimiter to be used for column separation – changes specs! (default: `None`)
379            linebreak (str): linebreak to be used in CSV representation – changes specs! (default: `None`)
380            backupExisting (bool): if `True` and file at filepath already exists, it will be renamed
381                                   (timestamp appended) and the new file will be written out in place (default: `False`)
382        """
383        if delimiter != None:
384            self.setSpec("delimiter", delimiter)
385
386        if linebreak != None:
387            self.setSpec("linebreak", linebreak)
388
389        csv = str(self)
390
391        if backupExisting and os.path.isfile(filepath):
392            mtime = os.path.getmtime(filepath)
393            ts = datetime.fromtimestamp(mtime).strftime("_%Y%m%d-%H%M%S")
394            path, file = os.path.split(filepath)
395            file = list(file.rpartition("."))
396            if file[0] == "":
397                file.append(ts)
398            else:
399                file.insert(1, ts)
400
401            buFilepath = os.path.join(path, "".join(file))
402            os.rename(filepath, buFilepath)
403
404        with open(filepath, "w") as csv_file:
405            csv_file.write(csv)
