 26            This is a single representation of one metric.

 28
 29            def __init__(self, name, value, labels={}):
 30                """initialize the metric instance
 31
 32                Args:
 33                    value (mixed): actual value of the metric instance
 34                    labels (dict): set of labels for the metric instance (default: `{}`)
 35                """
 36                self.setName(name)
 37                self.setValue(value)
 38                self.labels = labels
 39
 40            def __eq__(self, other):
 41                """check equality even down to the metrics value
 42
 43                Args:
 44                    other (MetricInstance): metric instance to compare with
 45                """
 46                return self.same(other) and self.value == other.value
 47
 48            def setName(self, name):
 49                """set / change name
 50
 51                Args:
 52                    name (str): name to be set
 53                """
 54                self.name = name
 55
 56            def setValue(self, value):
 57                """set / change value
 58
 59                Args:
 60                    value (mixed / number): value to be set for Metric
 61                """
 62                self.value = value
 63
 64            def same(self, other):
 65                """check if metric instance equals other metric instance
 66
 67                ATTENTION: this method does not (!) compare the actual value of the instance.
 68                It only checks in the manner of Prometheus metrics if the instances have the same
 69                set of labels and if so, they are assumed to be equal, so the new metric can replace
 70                the old one.
 71
 72                Args:
 73                    other (MetricInstance): metric instance to compare with
 74
 75                Returns:
 76                    bool: `True` if label set is same, `False` if not or other typed object given.
 77                """
 78                if type(self) == type(other):
 79                    return (self.labels == other.labels) and (self.name == other.name)
 80                else:
 81                    return False

 83        def __init__(self, name, helpText=None, metricType="gauge"):
 84            """initialize Metric by name
 85
 86            Args:
 87                name (str): name of metric to be used
 88                helpText (str): short description about the metric identified by name
 89                metricType (str): type of metric to be used (default: `gauge`)
 90            """
 91            self.instances = []
 92            self.setName(name)
 93            self.setHelp(helpText)
 94            self.setType(metricType)

 97            """change name

 99            Args:

103            for i in self.instances:
104                i.setName(name)

107            """change help information about metric
108
109            Args:
110                helpText (str): help information about the metrics
111            """
112            self.helpText = helpText

114        def setType(self, metricType):
115            """change metric type
116
117            Args:
118                metricType (str): type to change metric to

120            self.type = metricType

122        def addMetric(self, value, labels={}):
123            """add a metric instance
124
125            Only a single metric with the same label set can be contained, so
126            those “duplicates” will overwrite existing ones.
127
128            Args:
129                value (mixed): value of metric instance
130                labels (mixed): labels that will identify instance
131            """
132            m = self.MetricInstance(self.name, value, labels)
133            valueChanged = False
134            for i, e in enumerate(self.instances):
135                if e.same(m):
136                    logger.debug(
137                        f"Update value of a `{self.name}` metric from `{e.value}` to `{value}`."
138                    )
139                    e.setValue(value)
140                    valueChanged = True
141                    break
142            if not valueChanged:
143                self.instances.append(m)

154                    "help": self.helpText,
155                    "instances": self.instances,

159    def __init__(self):
160        """create set of metrics"""
161        self.metrics = {}
162
163    def addMetric(
164        self, metricName, value=None, labels={}, helpText=None, metricType=None
165    ):
166        """add a (new) metric instance
167
168        Args:
169            metricName (str): name of metric to be added
170            value (mixed): value of actual metric set (default: `None`)
171            labels (dict): labels to be set for actual metric set – they identify a metric! (default: `{}`)
172            helpText (str): help information for the metric collection of name metricName (default: `None`)
173            metricType (str): type of metric to be used – see https://prometheus.io/docs/concepts/metric_types/
174                              (default: None, will default to `gauge` on creation)
175        """
176        if not metricName in self.metrics:
177            if metricType == None:
178                metricType = "gauge"
179                logger.debug(
180                    f"Defaulting metric type to `{metricType}` for new created metric `{metricName}`."
181                )
182            if helpText == None:
183                logger.warning(
184                    f"No help information passed for new metric `{metricName}`!"
185                )
186            self.metrics[metricName] = self.Metric(
187                name=metricName, helpText=helpText, metricType=metricType
188            )
189            logger.debug(f"Added new metric `{metricName}`")
190        else:
191            if helpText != None:
192                self.setHelp(metricName, helpText)
193                logger.info(f"Changed help for metric `{metricName}`.")
194            if metricType != None:
195                self.setType(metricName, metricType)
196                logger.warning(f"Changed type for metric `{metricName}`.")
197        if value != None:
198            self.metrics[metricName].addMetric(value, labels)
199
200    def renameMetrics(self, old_name, new_name):
201        """rename a bunch of metrics
202
203        Args:
204            old_name (str): old name to be renamed
205            new_name (str): new name to be used
206        """
207        self.metrics[old_name].setName(new_name)
208        self.metrics[new_name] = self.metrics[old_name]
209        self.metrics.pop(old_name)
210
211    def setHelp(self, metricName, helpText):
212        """change help for metric
213
214        Args:
215            metricName (str): name of metric to set the help information
216            helpText (str): help information about what the metric is to say
217        """
218        self.metrics[metricName].setHelp(helpText)
219
220    def setType(self, metricName, metricType="gauge"):
221        """change metric type
222
223        Args:
224            metricName (str): name of metric to set type for
225            metricType (str): type to be set for metric (default: `"gauge"`)
226        """
227        self.metrics[metricName].setType(metricType)
228
229    def prepare(self):
230        """Helper function to transform list of metric objects of class above into usable object
231
232        Returns:
233            dict: dictionary to be pushed into template of this module to be rendered
234        """
235        finalized = {}
236        for m in self.metrics.values():
237            finalized.update(m.representation())
238        return finalized
239
240    def __str__(self):
241        """string representation
242
243        The string representation of a metrics collection is meant to be fetched by
244        Prometheus or to be pushed to a Prometheus Push Gateway.
245
246        Returns:
247            str: prometheus metrics data
248        """
249        file_loader = FileSystemLoader(
250            os.path.join(os.path.dirname(os.path.abspath(__file__)), "tpl")
251        )
252        env = Environment(loader=file_loader)
253        template = env.get_template("metrics.j2")
254
255        output = template.render(metrics=self.prepare()).strip() + "\n"
256        return output

123        def addMetric(self, value, labels={}):
124            """add a metric instance
125
126            Only a single metric with the same label set can be contained, so
127            those “duplicates” will overwrite existing ones.
128
129            Args:
130                value (mixed): value of metric instance
131                labels (mixed): labels that will identify instance
132            """
133            m = self.MetricInstance(self.name, value, labels)
134            valueChanged = False
135            for i, e in enumerate(self.instances):
136                if e.same(m):
137                    logger.debug(
138                        f"Update value of a `{self.name}` metric from `{e.value}` to `{value}`."
139                    )
140                    e.setValue(value)
141                    valueChanged = True
142                    break
143            if not valueChanged:
144                self.instances.append(m)
