PyHelpers by @macwinnie

This is the documentation overview for PyHelpers project by @macwinnie.

To install this package, please run:

pip install macwinnie-pyhelpers

Usage could then e.g. be: 

from macwinnie_pyhelpers.PyHistory import printHistory
saveHistory("path/to/historyfile.txt")
printHistory()

from macwinnie_pyhelpers.CSV import CSV
csv = CSV()
csv.loadData([
    {
        "col1": "val_row1_col1",
        "col2": "val_row1_col2",
        "col3": "val_row1_col3",
        "col4": "val_row1_col4",
    },
    {
        "col1": "val_row2_col1",
        "col2": "val_row2_col2",
        "col3": "val_row2_col3",
        "col4": "val_row2_col4",
    }
], kvDict=True)
print(bool(len(csv) == 2))

import macwinnie_pyhelpers.SententialLogic as SL
from macwinnie_pyhelpers.Version import Version
SL.Sentence("1.2.9 >= 1.2.4 <= 1.2.5", interpretingExample=Version("1.2.3")) # == True
